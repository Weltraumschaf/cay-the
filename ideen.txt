.methodName(arg1, arg2, ..., argN)
    = call("methodName", arg1, arg2, ..., argN)
obj + 5
    = obj.add(5)
    = obj.call("add", 5)
obj - 5
    = obj.minus(5)
    = obj.call("minus", 5)
obj * 5
    = obj.times(5)
    = obj.call("times", 5)
obj / 5
    = obj.divide(5)
    = obj.call("divide", 5)

3 + 4 * 5 - 6
    = 3 + (4 * 5) - 6
    = 3 + 4.times(5) - 6
    = 3.add(4.times(5)).minus(6)
