\subsection*{Source}

A source file \texttt{MyModule.ct} may look like:

\begin{verbatim}
var integer foo = 23
var integer bar // is 0 by default
var integer baz = foo + bar

const snafu = "Hello !"

function doIt(integer i) {
    ...
}

function integer doWhat(integer i, integer j) {
    return i + j
}

var result = doWhat(foo, 42)	
\end{verbatim}
