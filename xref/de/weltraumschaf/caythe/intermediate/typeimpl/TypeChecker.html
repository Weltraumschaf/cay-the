50      public static boolean areAssignmentCompatible(TypeSpecification targetType, TypeSpecification valueType) {
51          if (null == targetType || null == valueType) {
52              return false;
53          }
54  
55          targetType = targetType.baseType();
56          valueType  = valueType.baseType();
57          boolean compatible = false;
58  
59          // Identical types
60          if (targetType == valueType) {
61              compatible = true;
62          }
63          // real := integer
64          else if (isReal(targetType) && isInteger(valueType)) {
65              compatible = true;
66          }
67          // string := string
68          else {
69              compatible = targetType.isPascalString() && valueType.isPascalString();
70          }
71  
72          return compatible;
73      }
74  
75      public static boolean areComparisonCompatible(TypeSpecification type1, TypeSpecification type2) {
76          if (null == type1 || null == type2) {
77              return false;
78          }
79  
80          type1 = type1.baseType();
81          type2 = type2.baseType();
82          TypeForm form = type1.getForm();
83          boolean compatible = false;
84  
85          if (type1 == type2 && (form == SCALAR || form == ENUMERATION)) {
86              compatible = true;
87          }
88          // One integer and one real.
89          else if (isAtLeastOneReal(type1, type2)) {
90              compatible = true;
91          }
92          // Two strings
93          else {
94              compatible = type1.isPascalString() && type2.isPascalString();
95          }
96  
97          return compatible;
98      }
