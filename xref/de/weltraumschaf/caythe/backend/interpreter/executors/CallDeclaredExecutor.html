24  public class CallDeclaredExecutor extends CallExecutor {

26      public CallDeclaredExecutor(Executor parent) {
27          super(parent);

30      @Override
31      public Object execute(CodeNode node) {
32          SymbolTableEntry routineId = (SymbolTableEntry) node.getAttribute(ID);
33          ActivationRecord newAr = MemoryFactory.createActivationRecord(routineId);
34  
35          // Execute any actual parameters and initialize
36          // the formal parameters in the new activation record.
37          if (node.getChildren().size() > 0) {
38              CodeNode parmsNode = node.getChildren().get(0);
39              ArrayList<CodeNode> actualNodes = parmsNode.getChildren();
40              ArrayList<SymbolTableEntry> formalIds =
41                  (ArrayList<SymbolTableEntry>) routineId.getAttribute(ROUTINE_PARAMS);
42              executeActualParms(actualNodes, formalIds, newAr);
43          }
44  
45          // Push the new activation record.
46          runtimeStack.push(newAr);
47  
48          sendCallMessage(node, routineId.getName());
49  
50          // Get the root node of the routine's intermediate code.
51          Code iCode = (Code) routineId.getAttribute(ROUTINE_INTERMEDIATE_CODE);
52          CodeNode rootNode = iCode.getRoot();
53  
54          // Execute the routine.
55          StatementExecutor statementExecutor = new StatementExecutor(this);
56          Object value = statementExecutor.execute(rootNode);
57  
58          // Pop off the activation record.
59          runtimeStack.pop();
60  
61          sendReturnMessage(node, routineId.getName());
62          return value;
63      }
64  
65      private void executeActualParms(ArrayList<CodeNode> actualNodes,
66                                      ArrayList<SymbolTableEntry> formalIds,
67                                      ActivationRecord newAr)
68      {
69          ExpressionExecutor expressionExecutor = new ExpressionExecutor(this);
70          AssignmentExecutor assignmentExecutor = new AssignmentExecutor(this);
71  
72          for (int i = 0; i < formalIds.size(); ++i) {
73              SymbolTableEntry formalId = formalIds.get(i);
74              Definition formalDefn = formalId.getDefinition();
75              Cell formalCell = newAr.getCell(formalId.getName());
76              CodeNode actualNode = actualNodes.get(i);
77  
78              // Value parameter.
79              if (formalDefn == VALUE_PARAM) {
80                  TypeSpecification formalType = formalId.getTypeSpecification();
81                  TypeSpecification valueType  = actualNode.getTypeSpecification().baseType();
82                  Object value = expressionExecutor.execute(actualNode);
83  
84                  assignmentExecutor.assignValue(actualNode, formalId,
85                                                 formalCell, formalType,
86                                                 value, valueType);
87              }
88              // VAR parameter.
89              else {
90                  Cell actualCell =
91                      (Cell) expressionExecutor.executeVariable(actualNode);
92                  formalCell.setValue(actualCell);
93              }
94          }
95      }
