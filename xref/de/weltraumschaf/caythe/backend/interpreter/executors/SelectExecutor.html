16  public class SelectExecutor extends StatementExecutor {
17  
18      public SelectExecutor(Executor parent) {
19          super(parent);
20      }
21  
22      // Jump table cache: entry key is a SELECT node,
23      //                   entry value is the jump table.
24      // Jump table: entry key is a selection value,
25      //             entry value is the branch statement.
26      private static HashMap<CodeNode, HashMap<Object, CodeNode>> jumpCache =
27          new HashMap<CodeNode, HashMap<Object, CodeNode>>();
28  
29      @Override
30      public Object execute(CodeNode node) {
31          // Is there already an entry for this SELECT node in the
32          // jump table cache? If not, create a jump table entry.
33          HashMap<Object, CodeNode> jumpTable = jumpCache.get(node);
34          if (jumpTable == null) {
35              jumpTable = createJumpTable(node);
36              jumpCache.put(node, jumpTable);
37          }
38  
39          // Get the SELECT node's children.
40          ArrayList<CodeNode> selectChildren = node.getChildren();
41          CodeNode exprNode = selectChildren.get(0);
42  
43          // Evaluate the SELECT expression.
44          ExpressionExecutor expressionExecutor = new ExpressionExecutor(this);
45          Object selectValue = expressionExecutor.execute(exprNode);
46  
47          // If there is a selection, execute the SELECT_BRANCH's statement.
48          CodeNode statementNode = jumpTable.get(selectValue);
49          if (statementNode != null) {
50              StatementExecutor statementExecutor = new StatementExecutor(this);
51              statementExecutor.execute(statementNode);
52          }
53  
54          ++executionCount;  // count the SELECT statement itself
55          return null;
56      }
57  
58      private HashMap<Object, CodeNode> createJumpTable(CodeNode node) {
59          HashMap<Object, CodeNode> jumpTable = new HashMap<Object, CodeNode>();
60  
61          // Loop over children that are SELECT_BRANCH nodes.
62          ArrayList<CodeNode> selectChildren = node.getChildren();
63          for (int i = 1; i < selectChildren.size(); ++i) {
64              CodeNode branchNode    = selectChildren.get(i);
65              CodeNode constantsNode = branchNode.getChildren().get(0);
66              CodeNode statementNode = branchNode.getChildren().get(1);
67  
68              // Loop over the constants children of the branch's CONSTANTS_NODE.
69              ArrayList<CodeNode> constantsList = constantsNode.getChildren();
70              for (CodeNode constantNode : constantsList) {
71                  // Create a jump table entry.
72                  // Convert a single-character string constant to a character.
73                  Object value = constantNode.getAttribute(VALUE);
74  
75                  if (constantNode.getType() == STRING_CONSTANT) {
76                      value = ((String)value).charAt(0);
77                  }
78  
79                  jumpTable.put(value, statementNode);
80              }
81          }
82  
83          return jumpTable;
84      }
