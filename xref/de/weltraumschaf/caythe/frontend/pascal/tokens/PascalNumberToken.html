27      /**
28       * Does not consider for leading signs only the absolute number will be extracted.
29       * Leading signs should be consumed as {@link PascalSpecialSymbolToken}.
30       *
31       * @param textBuffer
32       * @throws Exception
33       */
34      protected void extractNumber(StringBuilder textBuffer) throws Exception {
35          String wholeDigits    = null;  // digits before the decimal point
36          String fractionDigits = null;  // digits after the decimal point
37          String exponentDigits = null;  // exponent digits
38          char exponentSign     = '+';   // exponent sign '+' or '-'
39          boolean sawDotDot     = false; // true if saw .. token
40          char currentChar;              // current character
41  
42          type = INTEGER;  // assume INTEGER token type for now
43  
44          // Extract the digits of the whole part of the number.
45          wholeDigits = unsignedIntegerDigits(textBuffer);
46          if (type == ERROR) {
47              return;
48          }
49  
50          // Is there a . ?
51          // It could be a decimal point or the start of a .. token.
52          currentChar = currentChar();
53  
54          if (currentChar == '.') {
55              if (peekChar() == '.') {
56                  sawDotDot = true;  // it's a ".." token, so don't consume it
57              }
58              else {
59                  type = REAL;  // decimal point, so token type is REAL
60                  textBuffer.append(currentChar);
61                  nextChar();  // consume decimal point
62  
63                  // Collect the digits of the fraction part of the number.
64                  fractionDigits = unsignedIntegerDigits(textBuffer);
65  
66                  if (ERROR == type) {
67                      return;
68                  }
69              }
70          }
71  
72          // Is there an exponent part?
73          // There cannot be an exponent if we already saw a ".." token.
74          currentChar = currentChar();
75  
76          if (!sawDotDot && ((currentChar == 'E') || (currentChar == 'e'))) {
77              type = REAL;  // exponent, so token type is REAL
78              textBuffer.append(currentChar);
79              currentChar = nextChar();  // consume 'E' or 'e'
80  
81              // Exponent sign?
82              if ((currentChar == '+') || (currentChar == '-')) {
83                  textBuffer.append(currentChar);
84                  exponentSign = currentChar;
85                  nextChar();  // consume '+' or '-'
86              }
87  
88              // Extract the digits of the exponent.
89              exponentDigits = unsignedIntegerDigits(textBuffer);
90          }
91  
92          // Compute the value of an integer number token.
93          if (INTEGER == type) {
94              int integerValue = computeIntegerValue(wholeDigits);
95  
96              if (ERROR != type) {
97                  value = new Integer(integerValue);
98              }
99          }
100         // Compute the value of a real number token.
101         else if (REAL == type) {
102             float floatValue = computeFloatValue(
103 		wholeDigits,
104 		fractionDigits,
105 		exponentDigits,
106 		exponentSign
107 	    );
108 
109             if (ERROR != type) {
110                 value = new Float(floatValue);
111             }
112         }
113     }

115     /**
116      * Returns digit characters from the passed in {@link StringBuilder} until first non digit
117      * character.
118      *
119      * Digit characters are all characters true for {@link Character#isDigit(char)}.
120      * There must be at least one digit character, unless the token is marked
121      * erroneous and the method returns null.
122      *
123      * @param textBuffer
124      * @return
125      * @throws Exception
126      */
127     private String unsignedIntegerDigits(StringBuilder textBuffer) throws Exception {
128         char currentChar = currentChar();
129 
130         // Must have at least one digit.
131         if (!Character.isDigit(currentChar)) {
132             type  = ERROR;
133             value = INVALID_NUMBER;
134             return null;
135         }
136 
137         // Extract the digits.
138         StringBuilder digits = new StringBuilder();
139 
140         while (Character.isDigit(currentChar)) {
141             textBuffer.append(currentChar);
142             digits.append(currentChar);
143             currentChar = nextChar();  // consume digit
144         }
145 
146         return digits.toString();
147     }

149     private int computeIntegerValue(String digits) {
150         try {
151             return NumberComputer.computeIntegerValue(digits);
152         }
153         catch (NumberComputer.RangeException ex) {
154             type  = ERROR;
155             value = RANGE_INTEGER;
156             return 0;
157         }
158     }

160     private float computeFloatValue(String wholeDigits, String fractionDigits,

163         try {

166         catch (NumberComputer.RangeException ex) {

168             value = RANGE_REAL;
169             return 0.0f;
