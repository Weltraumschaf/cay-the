16  public class TypeSpecificationParser extends PascalTopDownParser {
17  
18      public TypeSpecificationParser(PascalTopDownParser parent) {
19          super(parent);
20      }
21  
22      // Synchronization set for starting a type specification.
23      static final EnumSet<PascalTokenType> TYPE_START_SET =
24          SimpleTypeParser.SIMPLE_TYPE_START_SET.clone();
25      static {
26          TYPE_START_SET.add(ARRAY);
27          TYPE_START_SET.add(RECORD);
28          TYPE_START_SET.add(SEMICOLON);
29      }
30  
31      public TypeSpecification parse(Token token) throws Exception {
32          // Synchronize at the start of a type specification.
33          token = synchronize(TYPE_START_SET);
34  
35          switch ((PascalTokenType) token.getType()) {
36              case ARRAY: {
37                  ArrayTypeParser arrayTypeParser = new ArrayTypeParser(this);
38                  return arrayTypeParser.parse(token);
39              }
40  
41              case RECORD: {
42                  RecordTypeParser recordTypeParser = new RecordTypeParser(this);
43                  return recordTypeParser.parse(token);
44              }
45  
46              default: {
47                  SimpleTypeParser simpleTypeParser = new SimpleTypeParser(this);
48                  return simpleTypeParser.parse(token);
49              }
50          }
51      }
