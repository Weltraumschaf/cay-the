25  public class ForStatementParser extends StatementParser {
26  
27      public ForStatementParser(PascalTopDownParser parent) {
28          super(parent);
29      }
30      // Synchronization set for TO or DOWNTO.
31      private static final EnumSet<PascalTokenType> TO_DOWNTO_SET =
32              ExpressionParser.EXPRESSION_START_SET.clone();
33  
34      static {
35          TO_DOWNTO_SET.add(TO);
36          TO_DOWNTO_SET.add(DOWNTO);
37          TO_DOWNTO_SET.addAll(StatementParser.STATEMENT_FOLLOW_SET);
38      }
39      // Synchronization set for DO.
40      private static final EnumSet<PascalTokenType> DO_SET =
41              StatementParser.STATEMENT_START_SET.clone();
42  
43      static {
44          DO_SET.add(DO);
45          DO_SET.addAll(StatementParser.STATEMENT_FOLLOW_SET);
46      }
47  
48      @Override
49      public CodeNode parse(Token token) throws Exception {
50          token = nextToken();  // consume the FOR
51          Token targetToken = token;
52  
53          // Create the loop COMPOUND, LOOP, and TEST nodes.
54          CodeNode compoundNode = CodeFactory.createCodeNode(COMPOUND);
55          CodeNode loopNode     = CodeFactory.createCodeNode(LOOP);
56          CodeNode testNode     = CodeFactory.createCodeNode(TEST);
57  
58          // Parse the embedded initial assignment.
59          AssignmentStatementParser assignmentParser =
60                  new AssignmentStatementParser(this);
61          CodeNode initAssignNode = assignmentParser.parse(token);
62          TypeSpecification controlType = initAssignNode != null
63                  ? initAssignNode.getTypeSpecification()
64                  : Predefined.undefinedType;
65  
66          // Set the current line number attribute.
67          setLineNumber(initAssignNode, targetToken);
68  
69          // Type check: The control variable's type must be integer
70          //             or enumeration.
71          if (!TypeChecker.isInteger(controlType)
72                  && ( controlType.getForm() != ENUMERATION )) {
73              errorHandler.flag(token, INCOMPATIBLE_TYPES, this);
74          }
75  
76          // The COMPOUND node adopts the initial ASSIGN and the LOOP nodes
77          // as its first and second children.
78          compoundNode.addChild(initAssignNode);
79          compoundNode.addChild(loopNode);
80  
81          // Synchronize at the TO or DOWNTO.
82          token = synchronize(TO_DOWNTO_SET);
83          TokenType direction = token.getType();
84  
85          // Look for the TO or DOWNTO.
86          if (( direction == TO ) || ( direction == DOWNTO )) {
87              token = nextToken();  // consume the TO or DOWNTO
88          } else {
89              direction = TO;
90              errorHandler.flag(token, MISSING_TO_DOWNTO, this);
91          }
92  
93          // Create a relational operator node: GT for TO, or LT for DOWNTO.
94          CodeNode relOpNode = CodeFactory.createCodeNode(direction == TO ? GT : LT);
95          relOpNode.setTypeSpecification(Predefined.booleanType);
96  
97          // Copy the control VARIABLE node. The relational operator
98          // node adopts the copied VARIABLE node as its first child.
99          CodeNode controlVarNode = initAssignNode.getChildren().get(0);
100         relOpNode.addChild(controlVarNode.copy());
101 
102         // Parse the termination expression. The relational operator node
103         // adopts the expression as its second child.
104         ExpressionParser expressionParser = new ExpressionParser(this);
105         CodeNode exprNode = expressionParser.parse(token);
106         relOpNode.addChild(exprNode);
107 
108         // Type check: The termination expression type must be assignment
109         //             compatible with the control variable's type.
110         TypeSpecification exprType = exprNode != null
111                 ? exprNode.getTypeSpecification()
112                 : Predefined.undefinedType;
113 
114         if (!TypeChecker.areAssignmentCompatible(controlType, exprType)) {
115             errorHandler.flag(token, INCOMPATIBLE_TYPES, this);
116         }
117 
118         // The TEST node adopts the relational operator node as its only child.
119         // The LOOP node adopts the TEST node as its first child.
120         testNode.addChild(relOpNode);
121         loopNode.addChild(testNode);
122 
123         // Synchronize at the DO.
124         token = synchronize(DO_SET);
125         if (token.getType() == DO) {
126             token = nextToken();  // consume the DO
127         } else {
128             errorHandler.flag(token, MISSING_DO, this);
129         }
130 
131         // Parse the nested statement. The LOOP node adopts the statement
132         // node as its second child.
133         StatementParser statementParser = new StatementParser(this);
134         loopNode.addChild(statementParser.parse(token));
135 
136         // Create an assignment with a copy of the control variable
137         // to advance the value of the variable.
138         CodeNode nextAssignNode = CodeFactory.createCodeNode(ASSIGN);
139         nextAssignNode.setTypeSpecification(controlType);
140         nextAssignNode.addChild(controlVarNode.copy());
141 
142         // Create the arithmetic operator node:
143         // ADD for TO, or SUBTRACT for DOWNTO.
144         CodeNode arithOpNode = CodeFactory.createCodeNode(direction == TO ? ADD : SUBTRACT);
145         arithOpNode.setTypeSpecification(Predefined.integerType);
146 
147         // The next operator node adopts a copy of the loop variable as its
148         // first child and the value 1 as its second child.
149         arithOpNode.addChild(controlVarNode.copy());
150         CodeNode oneNode = CodeFactory.createCodeNode(INTEGER_CONSTANT);
151         oneNode.setAttribute(VALUE, 1);
152         oneNode.setTypeSpecification(Predefined.integerType);
153         arithOpNode.addChild(oneNode);
154 
155         // The next ASSIGN node adopts the arithmetic operator node as its
156         // second child. The loop node adopts the next ASSIGN node as its
157         // third child.
158         nextAssignNode.addChild(arithOpNode);
159         loopNode.addChild(nextAssignNode);
160 
161         // Set the current line number attribute.
162         setLineNumber(nextAssignNode, targetToken);
163 
164         return compoundNode;
165     }
