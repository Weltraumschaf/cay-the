20  public class TypeDefinitionsParser extends DeclarationsParser {
21  
22      public TypeDefinitionsParser(PascalTopDownParser parent) {
23          super(parent);
24      }
25  
26      // Synchronization set for a type identifier.
27      private static final EnumSet<PascalTokenType> IDENTIFIER_SET =
28          DeclarationsParser.VAR_START_SET.clone();
29      static {
30          IDENTIFIER_SET.add(IDENTIFIER);
31      }
32  
33      // Synchronization set for the = token.
34      private static final EnumSet<PascalTokenType> EQUALS_SET =
35          ConstantDefinitionsParser.CONSTANT_START_SET.clone();
36      static {
37          EQUALS_SET.add(EQUALS);
38          EQUALS_SET.add(SEMICOLON);
39      }
40  
41      // Synchronization set for what follows a definition or declaration.
42      private static final EnumSet<PascalTokenType> FOLLOW_SET =
43          EnumSet.of(SEMICOLON);
44  
45      // Synchronization set for the start of the next definition or declaration.
46      private static final EnumSet<PascalTokenType> NEXT_START_SET =
47          DeclarationsParser.VAR_START_SET.clone();
48      static {
49          NEXT_START_SET.add(SEMICOLON);
50          NEXT_START_SET.add(IDENTIFIER);
51      }
52  
53      @Override
54      public SymbolTableEntry parse(Token token, SymbolTableEntry parentId) throws Exception {
55          token = synchronize(IDENTIFIER_SET);
56  
57          // Loop to parse a sequence of type definitions
58          // separated by semicolons.
59          while (token.getType() == IDENTIFIER) {
60              String name = token.getText().toLowerCase();
61              SymbolTableEntry typeId = symbolTableStack.lookupLocal(name);
62  
63              // Enter the new identifier into the symbol table
64              // but don't set how it's defined yet.
65              if (typeId == null) {
66                  typeId = symbolTableStack.enterLocal(name);
67                  typeId.appendLineNumber(token.getLineNumber());
68              }
69              else {
70                  errorHandler.flag(token, IDENTIFIER_REDEFINED, this);
71                  typeId = null;
72              }
73  
74              token = nextToken();  // consume the identifier token
75  
76              // Synchronize on the = token.
77              token = synchronize(EQUALS_SET);
78              if (token.getType() == EQUALS) {
79                  token = nextToken();  // consume the =
80              }
81              else {
82                  errorHandler.flag(token, MISSING_EQUALS, this);
83              }
84  
85              // Parse the type specification.
86              TypeSpecificationParser typeSpecificationParser =
87                  new TypeSpecificationParser(this);
88              TypeSpecification type = typeSpecificationParser.parse(token);
89  
90              // Set identifier to be a type and set its type specificationt.
91              if (typeId != null) {
92                  typeId.setDefinition(DefinitionImpl.TYPE);
93              }
94  
95              // Cross-link the type identifier and the type specification.
96              if ((type != null) && (typeId != null)) {
97                  if (type.getIdentifier() == null) {
98                      type.setIdentifier(typeId);
99                  }
100 
101                 typeId.setTypeSpecification(type);
102             }
103             else {
104                 token = synchronize(FOLLOW_SET);
105             }
106 
107             token = currentToken();
108             TokenType tokenType = token.getType();
109 
110             // Look for one or more semicolons after a definition.
111             if (tokenType == SEMICOLON) {
112                 while (token.getType() == SEMICOLON) {
113                     token = nextToken();  // consume the ;
114                 }
115             }
116 
117             // If at the start of the next definition or declaration,
118             // then missing a semicolon.
119             else if (NEXT_START_SET.contains(tokenType)) {
120                 errorHandler.flag(token, MISSING_SEMICOLON, this);
121             }
122 
123             token = synchronize(IDENTIFIER_SET);
124         }
125 
126         return null;
127     }
