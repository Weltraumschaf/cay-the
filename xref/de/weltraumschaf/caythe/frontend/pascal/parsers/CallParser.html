38      @Override
39      public CodeNode parse(Token token) throws Exception {
40          SymbolTableEntry pfId = symbolTableStack.lookup(token.getText().toLowerCase());
41          RoutineCode routineCode = (RoutineCode) pfId.getAttribute(ROUTINE_CODE);
42          StatementParser callParser;
43  
44          if (( routineCode == DECLARED ) || ( routineCode == FORWARD )) {
45              callParser = new CallDeclaredParser(this);
46          } else {
47              callParser = new CallStandardParser(this);
48          }
49  
50          return callParser.parse(token);
51      }

53      private static final EnumSet<PascalTokenType> COMMA_SET =
54              ExpressionParser.EXPRESSION_START_SET.clone();

56      static {
57          COMMA_SET.add(PascalTokenType.COMMA);
58          COMMA_SET.add(PascalTokenType.RIGHT_PAREN);

61      protected CodeNode parseActualParameters(Token token, SymbolTableEntry pfId, boolean isDecalred, boolean isReadReadln, boolean isWriteWriteln) throws Exception {
62          ExpressionParser            expressionParser = new ExpressionParser(this);
63          CodeNode                    parmsNode        = CodeFactory.createCodeNode(PARAMETERS);
64          ArrayList<SymbolTableEntry> formalParms      = null;
65          int paramCount = 0;
66          int paramIndex = -1;
67  
68          if (isDecalred) {
69              formalParms = (ArrayList<SymbolTableEntry>) pfId.getAttribute(ROUTINE_PARAMS);
70              paramCount  = formalParms != null
71                            ? formalParms.size()
72                            : 0;
73          }
74  
75          if (token.getType() != PascalTokenType.LEFT_PAREN) {
76              if (paramCount != 0) {
77                  errorHandler.flag(token, PascalErrorCode.WRONG_NUMBER_OF_PARAMS, this);
78              }
79  
80              return null;
81          }
82  
83          token = nextToken(); // consume opening (
84  
85          // Loop to parse each actual parameter.
86          while (token.getType() != PascalTokenType.RIGHT_PAREN) {
87              CodeNode actualNode = expressionParser.parse(token);
88              // Declared procedure or function: Check the number of actual
89              // parameters, and check each actual parameter against the
90              // corresponding formal parameter.
91              if (isDecalred) {
92                  if (++paramIndex < paramCount) {
93                      SymbolTableEntry formalId = formalParms.get(paramIndex);
94                      checkActualParameter(token, formalId, actualNode);
95                  } else if (paramIndex == paramCount) {
96                      errorHandler.flag(token, PascalErrorCode.WRONG_NUMBER_OF_PARAMS, this);
97                  }
98              }
99              // read or readln: Each actual variable must be a variable that is
100             //                 a scalar, boolean or subrange of integer,
101             else if (isReadReadln) {
102                 TypeSpecification type = actualNode.getTypeSpecification();
103                 TypeForm          form = type != null
104                                          ? type.getForm()
105                                          : null;
106 
107                 if (!( ( actualNode.getType() == CodeNodeTypeImpl.VARIABLE )
108                         && ( ( form == TypeFormImpl.SCALAR )
109                         || ( type == Predefined.booleanType )
110                         || ( ( form == TypeFormImpl.SUBRANGE )
111                         && ( type.baseType() == Predefined.integerType ) ) ) )) {
112                     errorHandler.flag(token, PascalErrorCode.INVALID_VAR_PARAM, this);
113                 }
114             }
115             // write or writeln: The type of each actual parameter must be a
116             //                   scalar, boolean, or a Pascal string. Parse any field
117             //                   width and precision.
118             else if (isWriteWriteln) {
119                 CodeNode exprNode = actualNode;
120                 actualNode = CodeFactory.createCodeNode(WRITE_PARAM);
121                 actualNode.addChild(exprNode);
122                 TypeSpecification type = exprNode.getTypeSpecification().baseType();
123                 TypeForm form = type.getForm();
124 
125                 if (!( ( form == TypeFormImpl.SCALAR ) || ( type == Predefined.booleanType )
126                         || ( type.isPascalString() ) )) {
127                     errorHandler.flag(token, PascalErrorCode.INCOMPATIBLE_TYPES, this);
128                 }
129 
130                 // Optional field width.
131                 token = currentToken();
132                 actualNode.addChild(parseWriteSpec(token));
133                 // Optional precision.
134                 token = currentToken();
135                 actualNode.addChild(parseWriteSpec(token));
136             }
137 
138             parmsNode.addChild(actualNode);
139             token = synchronize(COMMA_SET);
140             TokenType tokenType = token.getType();
141 
142             // Look for the comma.
143             if (tokenType == PascalTokenType.COMMA) {
144                 token = nextToken();
145             } else if (ExpressionParser.EXPRESSION_START_SET.contains((PascalTokenType)tokenType)) {
146                 errorHandler.flag(token, PascalErrorCode.MISSING_COMMA, this);
147             } else if (tokenType != PascalTokenType.RIGHT_PAREN) {
148                 token = synchronize(ExpressionParser.EXPRESSION_START_SET);
149             }
150         }
151 
152         token = nextToken(); // Consume closing )
153 
154         if (parmsNode.getChildren().isEmpty() || ( isDecalred && ( paramIndex != paramCount - 1 ) )) {
155             errorHandler.flag(token, PascalErrorCode.WRONG_NUMBER_OF_PARAMS, this);
156         }
157 
158         return parmsNode;
159     }

161     private void checkActualParameter(Token token, SymbolTableEntry formalId, CodeNode actualNode) {
162         Definition formalDefn = formalId.getDefinition();
163         TypeSpecification formalType = formalId.getTypeSpecification();
164         TypeSpecification actualType = actualNode.getTypeSpecification();
165 
166         // VAR parameter: The actual parameter must be a variable of the same
167         //                type as the formal parameter.
168         if (formalDefn == DefinitionImpl.VAR_PARAM) {
169             if (( actualNode.getType() != CodeNodeTypeImpl.VARIABLE )
170                     || ( actualType != formalType )) {
171                 errorHandler.flag(token, PascalErrorCode.INVALID_VAR_PARAM, this);
172             }
173         } // Value parameter: The actual parameter must be assignment-compatible
174         //                  with the formal parameter.
175         else if (!TypeChecker.areAssignmentCompatible(formalType, actualType)) {
176             errorHandler.flag(token, PascalErrorCode.INCOMPATIBLE_TYPES, this);
177         }
178     }

180     private CodeNode parseWriteSpec(Token token) throws Exception {
181         if (token.getType() == PascalTokenType.COLON) {
182             token = nextToken();  // consume :
183 
184             ExpressionParser expressionParser = new ExpressionParser(this);
185             CodeNode specNode = expressionParser.parse(token);
186 
187             if (specNode.getType() == INTEGER_CONSTANT) {
188                 return specNode;
189             }
190             else {
191                 errorHandler.flag(token, PascalErrorCode.INVALID_NUMBER, this);
192                 return null;
193             }
194         }
195         else {
196             return null;
197         }
198     }
