31      @Override
32      public CodeNode parse(Token token) throws Exception {
33          CodeNode callNode = CodeFactory.createCodeNode(CALL);
34          SymbolTableEntry pfId = symbolTableStack.lookup(token.getText().toLowerCase());
35          RoutineCode routineCode = (RoutineCode) pfId.getAttribute(ROUTINE_CODE);
36          callNode.setAttribute(ID, pfId);
37  
38          token = nextToken(); // Consume procedure or function idnetnifier.
39  
40          switch ((RoutineCodeImpl) routineCode) {
41              case READ:
42              case READLN:  return parseReadReadln(token, callNode, pfId);
43  
44              case WRITE:
45              case WRITELN: return parseWriteWriteln(token, callNode, pfId);
46  
47              case EOF:
48              case EOLN:    return parseEofEoln(token, callNode, pfId);
49  
50              case ABS:
51              case SQR:     return parseAbsSqr(token, callNode, pfId);
52  
53              case ARCTAN:
54              case COS:
55              case EXP:
56              case LN:
57              case SIN:
58              case SQRT:    return parseArctanConExpLnSinSqrt(token, callNode, pfId);
59  
60              case PRED:
61              case SUCC:    return parsePredSucc(token, callNode, pfId);
62  
63              case CHR:     return parseChr(token, callNode, pfId);
64              case ODD:     return parseOdd(token, callNode, pfId);
65              case ORD:     return parseOrd(token, callNode, pfId);
66  
67              case ROUND:
68              case TRUNC:   return parseRoundTrunc(token, callNode, pfId);
69  
70              default:      return null;
71          }
72      }

74      private CodeNode parseReadReadln(Token token, CodeNode callNode, SymbolTableEntry pfId) throws Exception {
75          CodeNode parmsNode = parseActualParameters(token, pfId, false, true, false);
76          callNode.addChild(parmsNode);
77  
78          // Read must have parameters
79          if ((pfId == Predefined.readId) && callNode.getChildren().isEmpty()) {
80              errorHandler.flag(token, PascalErrorCode.WRONG_NUMBER_OF_PARAMS, this);
81          }
82  
83          return callNode;
84      }

98      private CodeNode parseEofEoln(Token token, CodeNode callNode, SymbolTableEntry pfId) throws Exception {
99          CodeNode parmsNode = parseActualParameters(token, pfId, false, false, false);
100         callNode.addChild(parmsNode);
101 
102         // There should be no actual paremeters.
103         // The function return type is boolean.
104         if (checkParamCount(token, parmsNode, 0)) {
105             callNode.setTypeSpecification(Predefined.booleanType);
106         }
107 
108         return callNode;
109     }

111     private CodeNode parseAbsSqr(Token token, CodeNode callNode, SymbolTableEntry pfId) throws Exception {
112         CodeNode parmsNode = parseActualParameters(token, pfId, false, false, false);
113         callNode.addChild(parmsNode);
114 
115         // There should be one integer or real parameter.
116         // The function return type is the parameter type.
117         if (checkParamCount(token, parmsNode, 1)) {
118             TypeSpecification argType = parmsNode.getChildren().get(0).getTypeSpecification().baseType();
119 
120             if ((argType == Predefined.integerType) || (argType == Predefined.realType)) {
121                 callNode.setTypeSpecification(argType);
122             } else {
123                 errorHandler.flag(token, PascalErrorCode.INVALID_TYPE, this);
124             }
125         }
126 
127         return callNode;
128     }

244     private boolean checkParamCount(Token token, CodeNode parmsNode, int count) {
245         if ((parmsNode == null) && (count == 0)) {
246             return true;
247         }
248 
249         if (parmsNode.getChildren().size() == count) {
250             return true;
251         }
252 
253         errorHandler.flag(token, PascalErrorCode.WRONG_NUMBER_OF_PARAMS, this);
254         return false;
255     }
