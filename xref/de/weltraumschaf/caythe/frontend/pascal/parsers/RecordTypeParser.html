21  public class RecordTypeParser extends TypeSpecificationParser {
22  
23      public RecordTypeParser(PascalTopDownParser parent) {
24          super(parent);
25      }
26  
27      // Synchronization set for the END.
28      private static final EnumSet<PascalTokenType> END_SET =
29          DeclarationsParser.VAR_START_SET.clone();
30      static {
31          END_SET.add(END);
32          END_SET.add(SEMICOLON);
33      }
34  
35      @Override
36      public TypeSpecification parse(Token token) throws Exception {
37          TypeSpecification recordType = TypeFactory.createType(TypeFormImpl.RECORD);
38          token = nextToken();  // consume RECORD
39  
40          // Push a symbol table for the RECORD type specification.
41          recordType.setAttribute(TypeKeyImpl.RECORD_SYMBOL_TABLE, symbolTableStack.push());
42  
43          // Parse the field declarations.
44          VariableDeclarationsParser variableDeclarationsParser =
45              new VariableDeclarationsParser(this);
46          variableDeclarationsParser.setDefinition(DefinitionImpl.FIELD);
47          variableDeclarationsParser.parse(token, null);
48  
49          // Pop off the record's symbol table.
50          symbolTableStack.pop();
51  
52          // Synchronize at the END.
53          token = synchronize(END_SET);
54  
55          // Look for the END.
56          if (token.getType() == END) {
57              token = nextToken();  // consume END
58          }
59          else {
60              errorHandler.flag(token, MISSING_END, this);
61          }
62  
63          return recordType;
64      }
