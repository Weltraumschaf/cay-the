16  public class PascalScanner extends Scanner {

18      public PascalScanner(Source source) {
19          super(source);
20      }
21  
22      @Override
23      protected Token extractToken() throws Exception {
24          skipWhitespace();
25  
26          Token token;
27          char currentChar = currentChar();
28  
29          // Construct the next token.
30          // The current character determines the token type.
31          if (EOF == currentChar) {
32              token = new EofToken(getSource());
33          } else if (Character.isLetter(currentChar)) {
34              token = new PascalWordToken(getSource());
35          } else if (Character.isDigit(currentChar)) {
36              token = new PascalNumberToken(getSource());
37          } else if ('\'' == currentChar) {
38              token = new PascalStringToken(getSource());
39          } else if (PascalTokenType.SPECIAL_SYMBOLS.containsKey(Character.toString(currentChar))) {
40              token = new PascalSpecialSymbolToken(getSource());
41          } else {
42              token = new PascalErrorToken(getSource(), INVALID_CHARACTER, Character.toString(currentChar));
43          }
44  
45          token.extract();
46          return token;
47      }
48  
49      /**
50       * Skips white spaces and comments.
51       *
52       * @throws Exception
53       */
54      private void skipWhitespace() throws Exception {
55          char currentChar = currentChar();
56  
57          while (Character.isWhitespace(currentChar) || ( '{' == currentChar )) {
58              // Start of a comment?
59              if ('{' == currentChar) {
60                  do {
61                      currentChar = nextChar(); // Consume comment chars.
62                  } while (( '}' != currentChar ) && ( EOF != currentChar ));
63  
64                  // Found closing '}'?
65                  if ('}' == currentChar) {
66                      currentChar = nextChar(); // Consumes the '}'.
67                  }
68              } // Not a comment.
69              else {
70                  currentChar = nextChar();
71              }
72          }
73      }
