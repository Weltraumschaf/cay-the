15  public class FrontendFactory {
16  
17      /**
18       * Creates the parser for the given language and source.
19       *
20       * @param lang
21       * @param source
22       * @return
23       * @throws Exception
24       */
25      public static Parser createParser(Language lang, Source source) throws IllegalArgumentException {
26          Scanner scanner = createScanner(lang, source);
27  
28          switch (lang) {
29              case PASCAL: {
30                  return new PascalTopDownParser(scanner);
31              }
32              case CAYTHE: {
33                  return new CayTheTopDownParser(scanner);
34              }
35          }
36  
37          throw new RuntimeException(String.format("FrontendFactory#createScanner() should throw an exception on invalid language '%s'!", lang));
38      }
39  
40      /**
41       * Creates the scanner for the given language and source.
42       * 
43       * @param lang
44       * @param source
45       * @return
46       * @throws IllegalArgumentException
47       */
48      private static Scanner createScanner(Language lang, Source source) throws IllegalArgumentException {
49          switch (lang) {
50              case PASCAL: {
51                  return new PascalScanner(source);
52              }
53              case CAYTHE: {
54                  return new CayTheScanner(source);
55              }
56              default: {
57                  throw new IllegalArgumentException(String.format("Unsupported language '%s'!", lang));
58              }
59          }
60      }
